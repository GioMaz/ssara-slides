\begin{definition}[Interference Graph]
    Graph $G = (V, E)$ such that \begin{itemize}
        \item $V$ is the set of variables;
        \item $E$ is the set of edges such that $\{ u, v \} \in E$, iff variables $u$ and $v$ interfere;
    \end{itemize}
\end{definition}

\begin{definition}[Single Static Assignment (SSA)]
    A program is in SSA form iff:
    \begin{itemize}
        \item Every variable is defined \textit{exactly} once;
        \item For every variable, its definition comes before its usages;
    \end{itemize}
\end{definition}
